```tsx
export const Typing = ({ session, text, speed = 10 }: Props) => {
  const items = text.split('').map((s, i) => (
    <Wait key={s + i} time={speed}>
      {s}
    </Wait>
  ));

  return <Session onDone={session ? session.next : undefined}>{items}</Session>;
};
```
